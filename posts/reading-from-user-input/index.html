Command line tooling

A 3rd party crate called clap is better at handling command line arguments. GitHub - clap-rs/clap: A full featured, fast Command Line Argument Parser for Rust

 1use clap::{App, Arg};
 2fn main() {
 3    let args = App::new("grep-lite")
 4        .version("0.1")
 5        .about("serach for patterns")
 6        .arg(
 7            Arg::with_name("pattern")
 8                .help("The pattern to search for")
 9                .takes_value(true)
10                .required(true),
11        )
12        .arg(
13            Arg::with_name("input")
14                .help("File to search")
15                .takes_value(true)
16                .required(true),
17        )
18        .get_matches();
19
20	  let pattern = args.value_of("pattern").unwrap();
21	  let input = args.value_of("input").unwrap();
22}

Reading file

Need to use to open the file first with File::open() and pass the untapped return value to BufReader::new().

 1use std::fs::File;
 2use std::io::prelude::*;
 3use std::io::BufReader;
 4
 5fn main() {
 6    let f = File::open(input).unwrap();
 7    let reader = BufReader::new(f);
 8    for line_ in reader.lines() {
 9        let line = line_.unwrap();
10    }
11}

reader.lines() provide a iterator to read lines from the buffered file.
